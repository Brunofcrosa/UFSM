#include<stdio.h>

int verificaPrimo(int *n){
	int i, cont = 0;
	for(i = 1; i <= *n; i++){
		if(*n % i == 0){
			cont++;
		}
	}
	if(cont == 2){
		return 1;
	}
	else{
		return 0;
	}
	
}

int main(){
	int n;
	int r;
	scanf("%d", &n);
	while (verificaPrimo(&n) != 1){
		n++;
	}
	printf("proximo primo %d", n);
	return 0;	
	
}
